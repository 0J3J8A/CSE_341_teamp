/* global use, db */
// MongoDB Playground
// Reviews collection for vacation destinations

const database = 'vacation_api';
const collection = 'reviews';

// Create a new database.
use(database);

// Create a new collection with validation rules.
db.createCollection(collection, {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["destination", "rating", "comment"],
      properties: {
        destination: {
          bsonType: "string",
          description: "Destination name is required"
        },
        rating: {
          bsonType: "int",
          minimum: 1,
          maximum: 5,
          description: "Rating must be between 1 and 5"
        },
        comment: {
          bsonType: "string",
          description: "Comment is required"
        },
        author: {
          bsonType: "string",
          description: "Author name, defaults to Anonymous"
        },
        createdAt: {
          bsonType: "date",
          description: "Auto-generated timestamp"
        },
        updatedAt: {
          bsonType: "date",
          description: "Auto-generated timestamp"
        }
      }
    }
  }
});
